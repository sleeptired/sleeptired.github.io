    {
        bool isMatch = false;
        if (name == recipes_[i].GetpotionName())
        {
            isMatch = true;
        }
        else
        {
            for (int j = 0; j < this->recipes_.size(); j++)
            {
                for (int z = 0; z < this->recipes_[j].Getingredients().size(); z++)
                {
                    if (this->recipes_[j].Getingredients()[z] == name)
                    {
                        isMatch = true;
                        break;
                    }
                }
            }

        }

        if (isMatch)
        {
            Recipes_Results.push_back(recipes_[i]);
        }
    }
